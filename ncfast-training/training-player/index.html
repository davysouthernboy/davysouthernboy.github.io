  <script>

    const PASSING_SCORE = 80;


    const modules = [

      {

            answer: 0

        ]

        title: "Module 3 - Basic Medicaid Application Entry",

        quiz: [

          {

            question: "At the end of a lesson, quizzes are useful because they:",

            choices: [

              "Slow everything down for no reason",

              "Reinforce key steps and understanding",

              "Replace policy manuals entirely",

              "Prevent all future mistakes"

            ],

            answer: 1

          }

    let currentModuleIndex = 0;

    let learnerName = "";


    const nameSection = document.getElementById("nameSection");

    const learnerNameInput = document.getElementById("learnerName");

    const startBtn = document.getElementById("startBtn");


    const trainingSection = document.getElementById("trainingSection");

    const lessonHeading = document.getElementById("lessonHeading");

    const moduleTitle = document.getElementById("moduleTitle");

    const lessonVideo = document.getElementById("lessonVideo");

    const quizSection = document.getElementById("quizSection");

    const quizForm = document.getElementById("quizForm");

    const submitQuizBtn = document.getElementById("submitQuizBtn");

    const quizResult = document.getElementById("quizResult");

    const nextLessonBtn = document.getElementById("nextLessonBtn");

    const retryLessonBtn = document.getElementById("retryLessonBtn");

    const progressBox = document.getElementById("progressBox");


    const certificateSection = document.getElementById("certificateSection");

    const certificateName = document.getElementById("certificateName");

    const certificateDate = document.getElementById("certificateDate");


    function saveProgress() {

      const progress = {

        learnerName,

        currentModuleIndex

      };

      localStorage.setItem("ncfastTrainingProgress", JSON.stringify(progress));

    }


    function loadProgress() {

      const raw = localStorage.getItem("ncfastTrainingProgress");

      if (!raw) return null;


      try {

        return JSON.parse(raw);

      } catch {

        return null;

      }

    }


    function renderProgress() {

      progressBox.innerHTML = `

        <strong>Learner:</strong> ${learnerName || "Not set"}<br>

        <strong>Progress:</strong> Module ${Math.min(currentModuleIndex + 1, modules.length)} of ${modules.length}

      `;

    }


    function loadModule(index) {

      const module = modules[index];

      lessonHeading.textContent = module.title;

      moduleTitle.textContent = module.title;

      lessonVideo.src = module.video;

      lessonVideo.load();


      quizSection.classList.add("hidden");

      quizResult.classList.add("hidden");

      nextLessonBtn.classList.add("hidden");

      retryLessonBtn.classList.add("hidden");


      renderQuiz(module.quiz);

      renderProgress();

      saveProgress();

    }


    function renderQuiz(questions) {

      quizForm.innerHTML = "";


      questions.forEach((q, qIndex) => {

        const wrapper = document.createElement("div");

        wrapper.className = "question";


        const prompt = document.createElement("p");

        prompt.textContent = `${qIndex + 1}. ${q.question}`;

        wrapper.appendChild(prompt);


        q.choices.forEach((choice, cIndex) => {

          const label = document.createElement("label");

          label.innerHTML = `

            <input type="radio" name="question-${qIndex}" value="${cIndex}">

            ${choice}

          `;

          wrapper.appendChild(label);

        });


        quizForm.appendChild(wrapper);

      });

    }


    function scoreQuiz() {

      const module = modules[currentModuleIndex];

      const questions = module.quiz;

      let correct = 0;


      questions.forEach((q, qIndex) => {

        const selected = document.querySelector(`input[name="question-${qIndex}"]:checked`);

        if (selected && Number(selected.value) === q.answer) {

          correct++;

        }

      });


      const percent = Math.round((correct / questions.length) * 100);

      return { correct, total: questions.length, percent };

    }


    function showCertificate() {

      trainingSection.classList.add("hidden");

      certificateSection.classList.remove("hidden");

      certificateName.textContent = learnerName;

      certificateDate.textContent = `Date: ${new Date().toLocaleDateString()}`;

      moduleTitle.textContent = "Training Complete";

    }


    startBtn.addEventListener("click", () => {

      const enteredName = learnerNameInput.value.trim();

      if (!enteredName) {

        alert("Please enter the learner's name.");

        return;

      }


      learnerName = enteredName;

      nameSection.classList.add("hidden");

      trainingSection.classList.remove("hidden");

      currentModuleIndex = 0;

      loadModule(currentModuleIndex);

    });


    lessonVideo.addEventListener("ended", () => {

      quizSection.classList.remove("hidden");

      quizSection.scrollIntoView({ behavior: "smooth" });

    });


    submitQuizBtn.addEventListener("click", () => {

      const result = scoreQuiz();


      quizResult.classList.remove("hidden");

      quizResult.innerHTML = `

        <strong>Score:</strong> ${result.correct} / ${result.total} (${result.percent}%)

      `;


      if (result.percent >= PASSING_SCORE) {

        quizResult.className = "status pass";

        quizResult.innerHTML += "<br><strong>Status:</strong> Passed";


        if (currentModuleIndex < modules.length - 1) {

          nextLessonBtn.classList.remove("hidden");

        } else {

          showCertificate();

        }

      } else {

        quizResult.className = "status fail";

        quizResult.innerHTML += `<br><strong>Status:</strong> Not yet passing (need ${PASSING_SCORE}%)`;

        retryLessonBtn.classList.remove("hidden");

      }

    });


    nextLessonBtn.addEventListener("click", () => {

      currentModuleIndex++;

      loadModule(currentModuleIndex);

      window.scrollTo({ top: 0, behavior: "smooth" });

    });


    retryLessonBtn.addEventListener("click", () => {

      loadModule(currentModuleIndex);

      window.scrollTo({ top: 0, behavior: "smooth" });

    });


    // Optional resume behavior

    window.addEventListener("load", () => {

      const saved = loadProgress();

      if (saved && saved.learnerName) {

        learnerNameInput.value = saved.learnerName;

      }

    });
